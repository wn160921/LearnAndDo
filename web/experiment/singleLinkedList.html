      void CreateList_L_Front(LinkList &L,ElemType a[],int n )
      {
          LinkList p; int i;
          L=(LinkList)malloc(sizeof(LNode));
          L->next=NULL;
          for(i=n-1;i>=0;i--)
          {
              p=(LinkList)malloc(sizeof(LNode));
              p->data=a[i];
              p->next=L->next;
              L->next=p;
          }
      }
